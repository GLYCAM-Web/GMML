00614     in >> std::setw(15) >> temp_force_constant
00615        >> std::setw(15) >> temp_phase
00616        >> std::setw(15) >> temp_periodicity;      /// Tokenize the rest of the line into the corresponding variables
00617 
00618     t.SetForceConstant(temp_force_constant);
00619     t.SetPhase(temp_phase);
00620     t.SetPeriodicity(temp_periodicity);
00621 
00622     if (in.fail())                              /// Invalid entry
00623         throw std::exception();
00624 
00625     t.SetFactor(iNotSet);                        /// Improper dihedral doesn't have factor
00626 
00627     if (line.size() > 60)                       /// Line has description
00628         dscr = line.substr(60);
00629 
00630     t.SetDscr(dscr);
00631     terms.push_back(t);
00632 
00633     scee = ProcessDoubleDihedralDescription(dscr, "SCEE");              /// Extract scee from the description column of the line
00634     scnb = ProcessDoubleDihedralDescription(dscr, "SCNB");              /// Extract scnb from the description column of the line
00635 
00636     for(unsigned int i = 0; i < types.size(); i++)
00637         Trim(types[i]);
00638     ParameterFileDihedral *dihedral = new ParameterFileDihedral(types, t, scee, scnb);
00639     while (dihedral->GetTerms().at(dihedral->GetTerms().size() - 1).GetPeriodicity() < 0)       /// Processing the following lines with the same dihedral;
00640         /// While the periodicity is negative the following lines are the same dihedrals with different attributes
00641     {
00642         getline(in_file, line);                     /// Read the following line with the same atom types in the dihedral
00643         line_number++;                              /// Increment the line counter
00644         ParameterFileDihedralTerm new_term;
00645 
00646         istringstream in2(line.substr(11));             /// Skip the first column of the line which is the same dihedral atom types
00647         in2 >> std::setw(15) >> temp_force_constant
00648             >> std::setw(15) >> temp_phase
00649             >> std::setw(15) >> temp_periodicity;  /// Tokenize the rest of the line into the corresponding variables
00650 
00651         new_term.SetForceConstant(temp_force_constant);
00652         new_term.SetPhase(temp_phase);
00653         new_term.SetPeriodicity(temp_periodicity);
00654 
00655         if (in2.fail())
00656         {
00657             throw ParameterFileProcessingException(line_number, "Error processing dihedral term");
00658         }
00659         if (line.size() > 60)                   /// Line has description
00660             dscr = line.substr(60);
00661 
00662         new_term.SetDscr(dscr);
00663         terms.push_back(new_term);
00664 
00665         dihedral->AddTerm(new_term);
00666     }
00667 
00668     vector<string> inverse_types(4);                /// Create an inverse vector of atom types for duplicate checking
00669     for(unsigned int i = 0; i < types.size(); i++)
00670     {
00671         inverse_types[i] = types[types.size() - (i+1)];
00672     }
00673 
00674     if(dihedrals_.find(types) == dihedrals_.end() && dihedrals_.find(inverse_types) == dihedrals_.end())
00675     {
00676         dihedral->SetIsImproper(true);                                 /// Set is_improper_ attribute to true; improper dihedral doesn't have factor
00677         if (dihedral->GetTypes().at(0) == "X" || dihedral->GetTypes().at(1) == "X" || dihedral->GetTypes().at(2) == "X" || dihedral->GetTypes().at(3) == "X")   ///  Check for generic dihedral
00678         {
00679             dihedral->SetIsGeneric(true);
00680         }
00681         else
00682         {
00683             dihedral->SetIsGeneric(false);
00684         }
00685         dihedrals_[dihedral->GetTypes()] = dihedral;
00686     }

00764 /// Process the potential parameter lines of the parameter file: Furthur information for atom types
00765 void ParameterFile::ProcessPotentialParameter(const std::string& line)
00766 {
00767     string type, dscr;
00768     double radius, depth;
00769 
00770     istringstream in(line);             /// Create an stream from the read line
00771     in >> type >> radius >> depth;      /// Tokenize the read line into the type, radius and well depth
00772 
00773     if (in.fail())                      /// Invalid entry
00774         throw std::exception();
00775 
00776     if (line.size() > 38)               /// Line has description
00777         dscr = line.substr(38);
00778 
00779     Trim(type);                         /// Remove spaces from the begining and the end of the string
00780     if (atom_types_.find(type) == atom_types_.end())        /// Check for the existing atom type in the map
00781     {
00782         atom_types_[type] = new ParameterFileAtom(type, dNotSet, dNotSet, radius, depth, dscr); /// Create a new entry in the map for a non-existing atom type in the map
00783     }
00784     else
00785     {
00786         atom_types_[type]->SetRadius(radius) ;            /// Update radius attribute of the existing atom type in the map
00787         atom_types_[type]->SetWellDepth(depth);         /// Update well depth attribute of the existing atom type in the map
00788         atom_types_[type]->SetMod4Dscr(dscr);           /// Update mod4 description attribute of the existing atom type in the map
00789     }
00790 
00791     vector<string>::const_iterator it;
00792     vector<string> equivalent_atoms;
00793     if (atom_types_.find(type) != atom_types_.end())    /// Check if the atom type in the map has equivalent list
00794         equivalent_atoms = atom_types_[type] -> GetEquivalentList();
00795     else
00796         return;
00797 
00798     for (it = equivalent_atoms.begin(); it != equivalent_atoms.end(); ++it)     /// For all atom types in the equivalent list of the atom set the corresponding attributes
00799     {
00800         if (atom_types_.find(*it) == atom_types_.end())         /// Check for the non-existing atom type in the map
00801         {
00802             atom_types_[*it] = new ParameterFileAtom(*it, dNotSet, dNotSet, radius, depth, dscr);       /// Create a new entry in the map for the non-existing atom type in the map

00809         }
00810     }

00813 /// Process the description part of the dihedral line to extract a double value with a key reference
00814 double ParameterFile::ProcessDoubleDihedralDescription(const std::string& dscr, const std::string& key)
00815 {
00816     double val;
00817     size_t pos = dscr.find(string(key + "="));      /// Find the starting position of the value of the given key in the given description
00818     if (pos == string::npos)                        /// Key is not found
00819         return dNotSet;
00820     std::istringstream ss(dscr.substr(pos + key.size() + 1));
00821     ss >> val;                                      /// Extract the value of the key from the found position
00822     if (ss.fail())
00823         return dNotSet;
00824     return val;
00825 }
